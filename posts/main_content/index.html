    Here is Fibonacci sequence code (get sequence number by index) in Kotlin:

    fun fibonacciSequenceFast(index: Int): Int {
    if (index <= 1){
        return index
    }

    var arr: Array<Int> = arrayOf(0, 1)
    var i = 1

    do {
        i++
        arr += (arr[i - 1] + arr[i - 2])
    } while (index != i)

    return arr[index]
}
